#include<iostream>
#include<cmath>
using namespace std;
int solve(int n , int m , int in){
    //Dá»«ng
    if(n==0) 
    {
        return 1;
    }
    
    int count = 0;

    for(int i = in + 1; i <= n ; i++){
        count+= solve(n - pow(i,m) , m , i);
    }
    return count; 
}

int main(){
    int n,m;
    cin >> n >> m ;
    cout << solve(n,m,0) << endl;
    return 0;
}
